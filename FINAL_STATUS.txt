#include "crawler.h"

int main() {
    WebCrawler crawler("mybot");
    
    RobotRule rule;
    rule.user_agents = {"*"};
    rule.disallows = {
        "/*.php",      // Block PHP files
        "/admin/*",    // Block admin
        "/*.pdf$"      // Block PDFs
    };
    rule.allows = {
        "/admin/public/*"  // Allow public
    };
    
    std::vector<RobotRule> rules = {rule};
    
    // Check paths
    crawler.is_path_allowed(rules, "/index.php");    // false
    crawler.is_path_allowed(rules, "/admin/secret");  // false
    crawler.is_path_allowed(rules, "/admin/public/file.html");  // true
}
